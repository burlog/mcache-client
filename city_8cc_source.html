92 // Hash 128 input bits down to 64 bits of output.
93 // This is intended to be a reasonably good hash function.
94 inline uint64 Hash128to64(const uint128& x) {
95  // Murmur-inspired hashing.
96  const uint64 kMul = 0x9ddfea08eb382d69ULL;
97  uint64 a = (Uint128Low64(x) ^ Uint128High64(x)) * kMul;
98  a ^= (a >> 47);
99  uint64 b = (Uint128High64(x) ^ a) * kMul;
100  b ^= (b >> 47);
101  b *= kMul;
102  return b;
103 }

179 // Some primes between 2^63 and 2^64 for various uses.
180 static const uint64 k0 = 0xc3a5c85c97cb3127ULL;
181 static const uint64 k1 = 0xb492b66fbe98f273ULL;
182 static const uint64 k2 = 0x9ae16a3b2f90404fULL;
183 static const uint64 k3 = 0xc949d7c7509e6557ULL;

185 // Bitwise right rotate. Normally this will compile to a single
186 // instruction, especially if the shift is a manifest constant.
187 static uint64 Rotate(uint64 val, int shift) {
188  // Avoid shifting by 64: doing so yields an undefined result.
189  return shift == 0 ? val : ((val >> shift) | (val << (64 - shift)));
190 }
191 
192 // Equivalent to Rotate(), but requires the second arg to be non-zero.
193 // On x86-64, and probably others, it's possible for this to compile
194 // to a single instruction if both args are already in registers.
195 static uint64 RotateByAtLeast1(uint64 val, int shift) {
196  return (val >> shift) | (val << (64 - shift));
197 }

207 static uint64 HashLen0to16(const char *s, size_t len) {
208  if (len > 8) {
209  uint64 a = Fetch64(s);
210  uint64 b = Fetch64(s + len - 8);
211  return HashLen16(a, RotateByAtLeast1(b + len, static_cast<int>(len))) ^ b;
212  }
213  if (len >= 4) {
214  uint64 a = Fetch32(s);
215  return HashLen16(len + (a << 3), Fetch32(s + len - 4));
216  }
217  if (len > 0) {
218  uint8 a = s[0];
219  uint8 b = s[len >> 1];
220  uint8 c = s[len - 1];
221  uint32 y = static_cast<uint32>(a) + (static_cast<uint32>(b) << 8);
222  uint32 z = static_cast<uint32>(len + (static_cast<uint32>(c) << 2));
223  return ShiftMix(y * k2 ^ z * k3) * k2;
224  }
225  return k2;
226 }

228 // This probably works well for 16-byte strings as well, but it may be overkill
229 // in that case.
230 static uint64 HashLen17to32(const char *s, size_t len) {
231  uint64 a = Fetch64(s) * k1;
232  uint64 b = Fetch64(s + 8);
233  uint64 c = Fetch64(s + len - 8) * k2;
234  uint64 d = Fetch64(s + len - 16) * k0;
235  return HashLen16(Rotate(a - b, 43) + Rotate(c, 30) + d,
236  a + Rotate(b ^ k3, 20) - c + len);
237 }

239 // Return a 16-byte hash for 48 bytes. Quick and dirty.
240 // Callers do best to use "random-looking" values for a and b.
241 static std::pair<uint64, uint64> WeakHashLen32WithSeeds(
242  uint64 w, uint64 x, uint64 y, uint64 z, uint64 a, uint64 b) {
243  a += w;
244  b = Rotate(b + a + z, 21);
245  uint64 c = a;
246  a += x;
247  a += y;
248  b += Rotate(a, 44);
249  return std::make_pair(a + z, b + c);
250 }

263 // Return an 8-byte hash for 33 to 64 bytes.
264 static uint64 HashLen33to64(const char *s, size_t len) {
265  uint64 z = Fetch64(s + 24);
266  uint64 a = Fetch64(s) + (len + Fetch64(s + len - 16)) * k0;
267  uint64 b = Rotate(a + z, 52);
268  uint64 c = Rotate(a, 37);
269  a += Fetch64(s + 8);
270  c += Rotate(a, 7);
271  a += Fetch64(s + 16);
272  uint64 vf = a + z;
273  uint64 vs = b + Rotate(a, 31) + c;
274  a = Fetch64(s + 16) + Fetch64(s + len - 32);
275  z = Fetch64(s + len - 8);
276  b = Rotate(a + z, 52);
277  c = Rotate(a, 37);
278  a += Fetch64(s + len - 24);
279  c += Rotate(a, 7);
280  a += Fetch64(s + len - 16);
281  uint64 wf = a + z;
282  uint64 ws = b + Rotate(a, 31) + c;
283  uint64 r = ShiftMix((vf + ws) * k2 + (wf + vs) * k0);
284  return ShiftMix(r * k0 + vs) * k2;
285 }

287 uint64 CityHash64(const char *s, size_t len) {
288  if (len <= 32) {
289  if (len <= 16) {
290  return HashLen0to16(s, len);
291  } else {
292  return HashLen17to32(s, len);
293  }
294  } else if (len <= 64) {
295  return HashLen33to64(s, len);
296  }
297 
298  // For strings over 64 bytes we hash the end first, and then as we
299  // loop we keep 56 bytes of state: v, w, x, y, and z.
300  uint64 x = Fetch64(s + len - 40);
301  uint64 y = Fetch64(s + len - 16) + Fetch64(s + len - 56);
302  uint64 z = HashLen16(Fetch64(s + len - 48) + len, Fetch64(s + len - 24));
303  std::pair<uint64, uint64> v = WeakHashLen32WithSeeds(s + len - 64, len, z);
304  std::pair<uint64, uint64> w = WeakHashLen32WithSeeds(s + len - 32, y + k1, x);
305  x = x * k1 + Fetch64(s);
306 
307  // Decrease len to the nearest multiple of 64, and operate on 64-byte chunks.
308  len = (len - 1) & ~static_cast<size_t>(63);
309  do {
310  x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
311  y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
312  x ^= w.second;
313  y += v.first + Fetch64(s + 40);
314  z = Rotate(z + w.first, 33) * k1;
315  v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
316  w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
317  std::swap(z, x);
318  s += 64;
319  len -= 64;
320  } while (len != 0);
321  return HashLen16(HashLen16(v.first, w.first) + ShiftMix(y) * k1 + z,
322  HashLen16(v.second, w.second) + x);
323 }

334 // A subroutine for CityHash128(). Returns a decent 128-bit hash for strings
335 // of any length representable in signed long. Based on City and Murmur.
336 static uint128 CityMurmur(const char *s, size_t len, uint128 seed) {
337  uint64 a = mc::Uint128Low64(seed);
338  uint64 b = mc::Uint128High64(seed);
339  uint64 c = 0;
340  uint64 d = 0;
341  signed long l = len - 16;
342  if (l <= 0) { // len <= 16
343  a = ShiftMix(a * k1) * k1;
344  c = b * k1 + HashLen0to16(s, len);
345  d = ShiftMix(a + (len >= 8 ? Fetch64(s) : c));
346  } else { // len > 16
347  c = HashLen16(Fetch64(s + len - 8) + k1, a);
348  d = HashLen16(b + len, c + Fetch64(s + len - 16));
349  a += d;
350  do {
351  a ^= ShiftMix(Fetch64(s) * k1) * k1;
352  a *= k1;
353  b ^= a;
354  c ^= ShiftMix(Fetch64(s + 8) * k1) * k1;
355  c *= k1;
356  d ^= c;
357  s += 16;
358  l -= 16;
359  } while (l > 0);
360  }
361  a = HashLen16(a, c);
362  b = HashLen16(d, b);
363  return uint128(a ^ b, HashLen16(b, a));
364 }

366 uint128 CityHash128WithSeed(const char *s, size_t len, uint128 seed) {
367  if (len < 128) {
368  return CityMurmur(s, len, seed);
369  }
370 
371  // We expect len >= 128 to be the common case. Keep 56 bytes of state:
372  // v, w, x, y, and z.
373  std::pair<uint64, uint64> v, w;
374  uint64 x = mc::Uint128Low64(seed);
375  uint64 y = mc::Uint128High64(seed);
376  uint64 z = len * k1;
377  v.first = Rotate(y ^ k1, 49) * k1 + Fetch64(s);
378  v.second = Rotate(v.first, 42) * k1 + Fetch64(s + 8);
379  w.first = Rotate(y + z, 35) * k1 + x;
380  w.second = Rotate(x + Fetch64(s + 88), 53) * k1;
381 
382  // This is the same inner loop as CityHash64(), manually unrolled.
383  do {
384  x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
385  y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
386  x ^= w.second;
387  y += v.first + Fetch64(s + 40);
388  z = Rotate(z + w.first, 33) * k1;
389  v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
390  w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
391  std::swap(z, x);
392  s += 64;
393  x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
394  y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
395  x ^= w.second;
396  y += v.first + Fetch64(s + 40);
397  z = Rotate(z + w.first, 33) * k1;
398  v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
399  w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
400  std::swap(z, x);
401  s += 64;
402  len -= 128;
403  } while (LIKELY(len >= 128));
404  x += Rotate(v.first + z, 49) * k0;
405  z += Rotate(w.first, 37) * k0;
406  // If 0 < len < 128, hash up to 4 chunks of 32 bytes each from the end of s.
407  for (size_t tail_done = 0; tail_done < len; ) {
408  tail_done += 32;
409  y = Rotate(x + y, 42) * k0 + v.second;
410  w.first += Fetch64(s + len - tail_done + 16);
411  x = x * k0 + w.first;
412  z += w.second + Fetch64(s + len - tail_done);
413  w.second += v.first;
414  v = WeakHashLen32WithSeeds(s + len - tail_done, v.first + z, v.second);
415  }
416  // At this point our 56 bytes of state should contain more than
417  // enough information for a strong 128-bit hash. We use two
418  // different 56-byte-to-8-byte hashes to get a 16-byte final result.
419  x = HashLen16(x, v.first);
420  y = HashLen16(y + z, w.first);
421  return uint128(HashLen16(x + v.second, w.second) + y,
422  HashLen16(x + w.second, y + v.second));
423 }

425 uint128 CityHash128(const char *s, size_t len) {
426  if (len >= 16) {
427  return CityHash128WithSeed(s + 16,
428  len - 16,
429  uint128(Fetch64(s) ^ k3,
430  Fetch64(s + 8)));
431  } else if (len >= 8) {
432  return CityHash128WithSeed(NULL,

434  uint128(Fetch64(s) ^ (len * k0),

437  return CityHash128WithSeed(s, len, uint128(k0, k1));

448 // Requires len >= 240.
449 static void CityHashCrc256Long(const char *s, size_t len,
450  uint32 seed, uint64 *result) {
451  uint64 a = Fetch64(s + 56) + k0;
452  uint64 b = Fetch64(s + 96) + k0;
453  uint64 c = result[0] = HashLen16(b, len);
454  uint64 d = result[1] = Fetch64(s + 120) * k0 + len;
455  uint64 e = Fetch64(s + 184) + seed;
456  uint64 f = seed;
457  uint64 g = 0;
458  uint64 h = 0;
459  uint64 i = 0;
460  uint64 j = 0;
461  uint64 t = c + d;
462 
463  // 240 bytes of input per iter.
464  size_t iters = len / 240;
465  len -= iters * 240;
466  do {
467 #define CHUNK(multiplier, z) \
468  { \
469  uint64 old_a = a; \
470  a = Rotate(b, 41 ^ z) * multiplier + Fetch64(s); \
471  b = Rotate(c, 27 ^ z) * multiplier + Fetch64(s + 8); \
472  c = Rotate(d, 41 ^ z) * multiplier + Fetch64(s + 16); \
473  d = Rotate(e, 33 ^ z) * multiplier + Fetch64(s + 24); \
474  e = Rotate(t, 25 ^ z) * multiplier + Fetch64(s + 32); \
475  t = old_a; \
476  } \
477  f = _mm_crc32_u64(f, a); \
478  g = _mm_crc32_u64(g, b); \
479  h = _mm_crc32_u64(h, c); \
480  i = _mm_crc32_u64(i, d); \
481  j = _mm_crc32_u64(j, e); \
482  s += 40
483 
484  CHUNK(1, 1); CHUNK(k0, 0);
485  CHUNK(1, 1); CHUNK(k0, 0);
486  CHUNK(1, 1); CHUNK(k0, 0);
487  } while (--iters > 0);
488 
489  while (len >= 40) {
490  CHUNK(k0, 0);
491  len -= 40;
492  }
493  if (len > 0) {
494  s = s + len - 40;
495  CHUNK(k0, 0);
496  }
497  j += i << 32;
498  a = HashLen16(a, j);
499  h += g << 32;
500  b += h;
501  c = HashLen16(c, f) + i;
502  d = HashLen16(d, e + result[0]);
503  j += e;
504  i += HashLen16(h, t);
505  e = HashLen16(a, d) + j;
506  f = HashLen16(b, c) + a;
507  g = HashLen16(j, i) + c;
508  result[0] = e + f + g + h;
509  a = ShiftMix((a + g) * k0) * k0 + b;
510  result[1] += a + result[0];
511  a = ShiftMix(a * k0) * k0 + c;
512  result[2] = a + result[1];
513  a = ShiftMix((a + e) * k0) * k0;
514  result[3] = a + result[2];
515 }

564 uint32_t city(const void *buf, std::size_t size, uint32_t seed = 0) {
565  // szn-ext: we add 'support' for seed
566  const char *s = reinterpret_cast<const char *>(buf);
567  std::size_t len = size;
568  uint64_t S = HashLen0to16(reinterpret_cast<const char *>(&seed), 4);
569 
570  if (len <= 32) {
571  if (len <= 16) {
572  return static_cast<uint32_t>(HashLen0to16(s, len) ^ S);
573  } else {
574  return static_cast<uint32_t>(HashLen17to32(s, len) + S);
575  }
576  } else if (len <= 64) {
577  return static_cast<uint32_t>(HashLen33to64(s, len) + S);
578  }
579 
580  // For strings over 64 bytes we hash the end first, and then as we
581  // loop we keep 56 bytes of state: v, w, x, y, and z.
582  uint64 x = Fetch64(s + len - 40);
583  uint64 y = Fetch64(s + len - 16) + Fetch64(s + len - 56);
584  uint64 z = HashLen16(Fetch64(s + len - 48) + len,
585  Fetch64(s + len - 24)) ^ S;
586  std::pair<uint64, uint64> v = WeakHashLen32WithSeeds(s + len - 64, len, z);
587  std::pair<uint64, uint64>
588  w = WeakHashLen32WithSeeds(s + len - 32, y + k1, x);
589  x = x * k1 + Fetch64(s);
590 
591  // Decrease len to the nearest multiple of 64, and operate on 64-byte chunks.
592  len = (len - 1) & ~static_cast<size_t>(63);
593  do {
594  x = Rotate(x + y + v.first + Fetch64(s + 8), 37) * k1;
595  y = Rotate(y + v.second + Fetch64(s + 48), 42) * k1;
596  x ^= w.second;
597  y += v.first + Fetch64(s + 40);
598  z = Rotate(z + w.first, 33) * k1;
599  v = WeakHashLen32WithSeeds(s, v.second * k1, x + w.first);
600  w = WeakHashLen32WithSeeds(s + 32, z + w.second, y + Fetch64(s + 16));
601  std::swap(z, x);
602  s += 64;
603  len -= 64;
604  } while (len != 0);
605  return static_cast<uint32_t>(
606  HashLen16(HashLen16(v.first, w.first) + ShiftMix(y) * k1 + z,
607  HashLen16(v.second, w.second) + x));
608 }
