114  bool is_dead() const { return shared->dead;}

120  bool callable() {
121  // if server is not marked dead return immediately
122  if (!shared->dead) return true;
123 
124  // check wether we shloud try make dead server alive
125  time_t now = ::time(0x0);
126  if (now < shared->restoration) return false;
127 
128  // if we don't get lock returns immediately
129  scope_guard_t<lock_t> guard(shared->lock);
130  if (!guard.try_lock()) return false;
131 
132  // we got lock, enlarge dead timeout and return true
133  shared->restoration = now + restoration_interval;
134  aux::log_server_raise_zombie(connections.server_name(),
135  restoration_interval);
136  return true;
137  }
138 
141  time_t lifespan() const {
142  time_t now = ::time(0x0);
143  if (!shared->restoration) return now;
144  return std::max(now - (shared->restoration - restoration_interval), 0l);
145  }

151  template <typename command_t>
152  typename command_t::response_t send(const command_t &command) {
153  // pick connection from pool of connections
154  typedef typename command_t::response_t response_t;
155  try {
156  // if command was finished successfuly then make server alive
157  connection_ptr_t connection = connections.pick();
158  proto::command_parser_t<connection_t> parser(*connection);
159  response_t response = parser.send(command);
160  shared->dead = false;
161  shared->fails = 0;
162 
163  // if command does not understand repsonse then does not return the
164  // connection to pool (the connection will be closed)
165  if (response.code() < proto::resp::error)
166  connections.push_back(connection);
167  return response;
168 
169  } catch (const io::error_t &e) {
170  // lock, destroy whole pool of connections and mark server as dead
171  scope_guard_t<lock_t> guard(shared->lock);
172  if (guard.try_lock()) {
173  if (++shared->fails >= fail_limit) {
174  connections.clear();
175  shared->restoration = ::time(0x0) + restoration_interval;
176  shared->dead = true;
177  aux::log_server_is_dead(connections.server_name(),
178  fail_limit,
179  restoration_interval,
180  e.what());
181  }
182  }
183  return response_t(proto::resp::io_error,
184  std::string("connection failed: ") + e.what());
185  }
186  // never reached
187  throw std::runtime_error(__PRETTY_FUNCTION__);
188  }

192  std::string state() const {
193  return aux::make_state_string(connections.server_name(),
194  connections.size(),
195  shared->restoration,
196  shared->fails,
