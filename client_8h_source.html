533 protected:
538  template <typename command_t>
539  typename command_t::response_t
540  run(const command_t &command, bool h404 = false) {
541  // we will never have this count of servers
542  typename pool_t::value_type
543  prev = std::numeric_limits<typename pool_t::value_type>::max();
544  // count of consequently continues
545  uint32_t conts = 0;
546  bool out_of_servers = true;
547 
548  // find callable server for given key
549  for (typename pool_t::const_iterator
550  iidx = pool.choose(command.key),
551  eidx = pool.end();
552  (iidx != eidx) && (conts < max_continues); ++iidx)
553  {
554  // skip previous key
555  if (*iidx == prev) continue;
556  prev = *iidx;
557 
558  // check if choosed server node is dead
559  typedef typename server_proxies_t::server_proxy_t server_proxy_t;
560  server_proxy_t &server = proxies[*iidx];
561  if (server.callable()) {
562  // send command to server and wait till response arrive
563  typename command_t::response_t response = server.send(command);
564  switch (response.code()) {
565  case proto::resp::io_error: break;
566  case proto::resp::not_found:
567  // if we got 404 for get command and we ask first server in
568  // the consistent hash ring and server have been recently
569  // restored so we can ask neighbour server for the data...
570  if (h404 && !conts && (server.lifespan() < h404_duration)) {
571  out_of_servers = false;
572  break;
573  }
574  // pass
575  default: return response;
576  }
577  }
578 
579  // can't be in for statement due to "skip previous key" continue
580  ++conts;
581  }
582  if (out_of_servers) throw out_of_servers_t();
583  return typename command_t::response_t(proto::resp::not_found);
584  }

589  const int64_t h404_duration;
