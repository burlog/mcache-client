151 template <typename hash_function_t>
152 class consistent_hashing_pool_t: public consistent_hashing_pool_base_t {
153 public:
155  using consistent_hashing_pool_base_t::value_type;
157  using consistent_hashing_pool_base_t::ring_t;
159  typedef consistent_hashing_pool_const_iterator const_iterator;

164  consistent_hashing_pool_t(const std::vector<std::string> &addresses,
165  const consistent_hashing_pool_config_t &
166  cfg = consistent_hashing_pool_config_t())
167  : ring(), hashf()
168  {
169  // at least one address must be supplied
170  if (addresses.empty()) throw std::out_of_range(__PRETTY_FUNCTION__);
171 
172  // create namespace ring
173  for (std::vector<std::string>::const_iterator
174  saddr = addresses.begin(),
175  iaddr = addresses.begin(),
176  eaddr = addresses.end();
177  iaddr != eaddr; ++iaddr)
178  {
179  // prepare index of server in addresses vector
180  uint32_t hash = 0;
181  uint32_t idx = static_cast<uint32_t>(std::distance(saddr, iaddr));
182 
183  // push server into namespace ring
184  for (uint32_t i = 0; i < cfg.virtual_nodes; ++i)
185  ring.insert(std::make_pair(hash = hashf(*iaddr, hash), idx));
186  }
187  }
188 
189  /* Returns iterator that points to the first index of usable server.
190  */
191  const_iterator choose(const std::string &key) const {
192  ring_t::const_iterator inode = ring.lower_bound(hashf(key));
193  return const_iterator(inode, ring.begin(), ring.end());
194  }
195 
196  /* Returns iterator that points to the first entry in namespace ring.
197  */
198  const_iterator begin() const {
199  return const_iterator(ring.begin(), ring.begin(), ring.end());
200  }
201 
204  const_iterator end() const { return const_iterator(ring.end());}
