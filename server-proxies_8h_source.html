137 template <
138  template <typename> class shared_templ_t,
139  typename server_proxy_type
140 > class server_proxies_t: public boost::noncopyable {
141 public:
142  // publish server proxy type
143  typedef server_proxy_type server_proxy_t;
144  // publish server proxy iterator
145  typedef const server_proxy_t *const_iterator;
146  // publish server proxy type
147  typedef typename server_proxy_t::server_proxy_config_type
148  server_proxy_config_t;
149 
152  server_proxies_t(const std::vector<std::string> &addresses,
153  const server_proxy_config_t &cfg = server_proxy_config_t())
154  : shared(addresses.size()), proxies(addresses.size()),
155  count(addresses.size())
156  {
157  for (std::vector<std::string>::const_iterator
158  iaddr = addresses.begin(),
159  saddr = addresses.begin(),
160  eaddr = addresses.end();
161  iaddr != eaddr; ++iaddr)
162  {
163  // initialize server proxy inplace via placement new operator
164  std::size_t i = std::distance(saddr, iaddr);
165  new (&proxies[i]) server_proxy_t(*iaddr, &shared[i], cfg);
166  }
167  }
168 
171  ~server_proxies_t() {
172  // if server_proxy_t throws than bad things can happen...
173  for (std::size_t i = 0; i < count; ++i) proxies[i].~server_proxy_t();
174  }
175 
178  server_proxy_t &operator[](std::size_t i) { return proxies[i];}
179 
180  /* Returns iterator that points to the first server proxy.
181  */
182  const server_proxy_t *begin() const { return &proxies[0];}
183 
186  const_iterator end() const { return &proxies[count - 1] + 1;}
187 
188 private:
189  // shortcuts
190  typedef aux::unitialized_array_t<server_proxy_t> proxies_t;
191  typedef shared_templ_t<typename server_proxy_t::shared_t> shared_array_t;
192 
193  shared_array_t shared;
194  proxies_t proxies;
195  std::size_t count;
196 };
