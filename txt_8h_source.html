157 template <typename parent_t, const char **name>

159 public:

163  name_injector(const type0_t &val0): parent_t(val0) {}
164 
167  template <typename type0_t, typename type1_t>

174  template <typename type0_t, typename type1_t, typename type2_t>

176  : parent_t(val0, val1, val2)
177  {}
178 
181  std::string serialize() const { return parent_t::serialize(*name);}
