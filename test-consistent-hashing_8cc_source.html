79  std::map<std::string, uint32_t> values;

84  // fill values storage with default setup

86  std::string key = test::rand_string();
87  uint32_t idx = *pool.choose(key);
88  if (values.insert(std::make_pair(key, idx)).second)
89  ++distribution[idx];
90  }

93  std::cout << _1 << '\n');

95  // remove server3 from pool

98  // check how many keys got different memcache after one server disappear
99  std::fill(distribution.begin(), distribution.end(), 0);
100  std::fill(migrated.begin(), migrated.end(), 0);
101  uint32_t changed = 0;
102  for (std::map<std::string, uint32_t>::const_iterator
103  ivalue = values.begin(), evalue = values.end();
104  ivalue != evalue; ++ivalue)
105  {
106  uint32_t idx = *pool.choose(ivalue->first);
107  changed += idx != ivalue->second;
108  if (idx != ivalue->second) ++migrated[idx];
109  ++distribution[idx];
110  }

113  << " keys shall migrate to other server: \n";
114  std::for_each(distribution.begin(), distribution.end(),
115  std::cout << _1 << '\n');
116  std::cout << "New destinations: \n";
117  std::for_each(migrated.begin(), migrated.end(),
118  std::cout << _1 << '\n');
