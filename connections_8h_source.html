223  connection_ptr_t pick() {
224  boost::mutex::scoped_lock guard(mutex);
225  if (stack.empty())
226  return boost::make_shared<connection_t>(addr, opts);
227  connection_ptr_t tmp = stack.top();
228  stack.pop();
229  return tmp;
230  }
231 
235  void push_back(connection_ptr_t &tmp) {
236  boost::mutex::scoped_lock guard(mutex);
237  if (stack.size() < opts.max_connections_in_pool) stack.push(tmp);
238  tmp.reset();
239  }
240 
243  std::size_t size() const {
244  boost::mutex::scoped_lock guard(mutex);
245  return stack.size();
246  }
247 
250  void clear() {
251  boost::mutex::scoped_lock guard(mutex);
252  while (!stack.empty()) stack.pop();
253  }
254 
257  const std::string &server_name() const { return addr;}
258 
259 protected:
260  std::string addr;
261  opts_t opts;
