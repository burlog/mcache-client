40 static std::string error_desc(const std::string &header) {
41  std::string::size_type space = header.find(' ');
42  if (space == std::string::npos) return std::string();
43  return boost::trim_copy(header.substr(space));
44 }

50 static retrieve_command_t::response_t
51 deserialize_value_resp(const std::string &header) {
52  // prepare response storage
53  typedef retrieve_command_t::response_t response_t;
54  std::string unused;
55  uint32_t flags = 0;
56  std::size_t bytes = 0;
57  uint64_t cas = 0;
58  DBG(DBG1, "Parsing header of get command: line=%s",
59  log::escape(header).c_str());
60 
61  // parse and return
62  std::istringstream is(header);
63  if ((is >> unused) && (is >> unused) && (is >> flags) && (is >> bytes)) {
64  is >> cas;
65  return response_t(flags, bytes + retrieve_command_t::footer_size, cas,
66  retrieve_command_t::set_body);
67  }
68  return response_t(resp::syntax, "invalid response: " + header);
69 }

73 command_t::response_t
74 command_t::deserialize_header(const std::string &header) const {
75  // try parse general errors (my childs ensures that header is not empty)
76  switch (header[0]) {
77  case 'E':
78  if (boost::starts_with(header, "ERROR"))
79  return response_t(resp::error, "syntax error");
80  break;
81  case 'C':
82  if (boost::starts_with(header, "CLIENT_ERROR"))
83  return response_t(resp::client_error, error_desc(header));
84  break;
85  case 'S':
86  if (boost::starts_with(header, "SERVER_ERROR"))
87  return response_t(resp::server_error, error_desc(header));
88  break;
89  default: break;
90  }
91 
92  // header does not recognized, return unrecognized
93  return response_t(resp::unrecognized, header);
94 }

96 retrieve_command_t::response_t
97 retrieve_command_t::deserialize_header(const std::string &header) const {
98  // reject empty response
99  if (header.empty()) return response_t(resp::empty, "empty response");
100 
101  // try parse retrieve responses
102  switch (header[0]) {
103  case 'E':
104  if (boost::starts_with(header, "END"))
105  return response_t(resp::not_found, "not found");
106  break;
107  case 'V':
108  if (boost::starts_with(header, "VALUE"))
109  return deserialize_value_resp(header);
110  break;
111  default: break;
112  }
113 
114  // header does not recognized, try global errors
115  return response_t(command_t::deserialize_header(header));
116 }

134 storage_command_t::response_t
135 storage_command_t::deserialize_header(const std::string &header) const {
136  // reject empty response
137  if (header.empty()) return response_t(resp::empty, "empty response");
138 
139  // try parse retrieve responses
140  switch (header[0]) {
141  case 'E':
142  if (boost::starts_with(header, "EXISTS"))
143  return response_t(resp::exists, "cas id expired");
144  break;
145  case 'N':
146  if (boost::starts_with(header, "NOT_FOUND"))
147  return response_t(resp::not_found, "cas id is invalid");
148  if (boost::starts_with(header, "NOT_STORED"))
149  return response_t(resp::not_stored, "key (does not) exist");
150  break;
151  case 'S':
152  if (boost::starts_with(header, "STORED"))
153  return response_t(resp::stored);
154  break;
155  default: break;
156  }
157 
158  // header does not recognized, try global errors
159  return command_t::deserialize_header(header);
160 }

162 std::string storage_command_t::serialize(const char *name) const {
163  aux::check_key(key);
164  // <command name> <key> <flags> <exptime> <bytes> [noreply]\r\n
165  // cas <key> <flags> <exptime> <bytes> <cas unique> [noreply]\r\n
166  std::ostringstream os;
167  os << name << ' '
168  << key << ' '
169  << opts.flags << ' '
170  << opts.expiration << ' '
171  << data.size();
172  if (opts.cas) os << ' ' << opts.cas;
173  os << header_delimiter() << data << header_delimiter();
174  return os.str();
175 }

177 incr_decr_command_t::response_t
178 incr_decr_command_t::deserialize_header(const std::string &header) const {
179  // reject empty response
180  if (header.empty()) return response_t(resp::empty, "empty response");
181 
182  // try parse retrieve responses
183  switch (header[0]) {
184  case 'T':
185  // fail response
186  if (boost::starts_with(header, "TOUCHED"))
187  return response_t(resp::touched);
188  break;
189  case 'N':
190  if (boost::starts_with(header, "NOT_FOUND"))
191  return response_t(resp::not_found, "key does not exist");
192  break;
193  case '0'...'9': {
194  // <value>\r\n - ok response
195  std::istringstream is(header);
196  uint64_t value;
197  if (is >> value)
198  return response_t(resp::ok, boost::trim_copy(header));
199  break;
200  }
201  default: break;
202  }
203 
204  // header does not recognized, try global errors
205  return command_t::deserialize_header(header);
206 }
