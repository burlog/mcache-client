96  // prepare deadline timer
97  deadline.expires_at(boost::posix_time::pos_infin);
98  handle_deadline();
99 
100  // resolve endpoints
101  std::pair<std::string, std::string> dest = parse_address(addr);
102  asio::ip::tcp::resolver::query query(dest.first, dest.second);
103  asio::ip::tcp::resolver::iterator
104  iendpoint = asio::ip::tcp::resolver(ios).resolve(query);
105 
106  // nice log
107  DBG(DBG2, "Resolved address of memcache server: server=%s, address=%s",
108  addr.c_str(),
109  iendpoint->endpoint().address().to_string().c_str());

111  // launch async connect
112  set_deadline(opts.timeouts.connect);
113  boost::system::error_code ec = asio::error::would_block;
114  asio::async_connect(socket, iendpoint, var(ec) = _1);
115  do { ios.run_one();} while (ec == asio::error::would_block);
116 
117  // check whether socket is connected
118  if (ec || !socket.is_open()) {
119  if (!timeouted) throw io::error_t(err::io_error, ec.message());
120  throw io::error_t(err::timeout, "can't connect due to timeout"

122  }

156  DBG(DBG1, "Schedule receiving buffer of data from server: "

183  std::string read(std::size_t count) {
184  DBG(DBG1, "Schedule receiving buffer of data from server: "
185  "count=%zd, buffered-size=%zd, buffered=%s",
186  count, input.size(), dump(input).c_str());
187 
188  // if there is sufficient bytes in buffer return immediately
189  if (count <= input.size()) return copy_n(input, count);
190 
191  // launch async read
192  set_deadline(opts.timeouts.read);
193  std::size_t transfer = count - input.size();
194  boost::system::error_code ec = asio::error::would_block;
195  asio::async_read(socket, input, asio::transfer_at_least(transfer),
196  var(ec) = _1);
197  do { ios.run_one();} while (ec == asio::error::would_block);
198 
199  // check whether data was read
200  if (ec) {
201  if (!timeouted) throw io::error_t(err::io_error, ec.message());
202  throw io::error_t(err::timeout, "can't read due to timeout"
203  ": dst=" + addr);
204  }
205  DBG(DBG3, "New buffer with data has been read from server: size=%zd",
206  input.size());
207  return copy_n(input, count);

214  const char *ptr = asio::buffer_cast<const char *>(stream.data());

233  void handle_deadline() {
234  // check whether timeout expired
235  if (deadline.expires_at() <= asio::deadline_timer::traits_type::now()) {
236  DBG(DBG3, "Connection timeouted: server=%s", addr.c_str());
237 
238  // close the socket due to timeout
239  boost::system::error_code ec;
240  socket.close(ec);
241  if (ec) {
242  LOG(ERR3, "Can't close socket: dest=%s, code=%d, error=%s",
243  addr.c_str(), ec.value(), ec.message().c_str());
244  }
245 
246  // disable timeout till new timeout will be set
247  deadline.expires_at(boost::posix_time::pos_infin);
248  timeouted = true;
249  }
250  // schedule new async op
251  deadline.async_wait(boost::bind(&this_t::handle_deadline, this));
252  }

400  packet_t receive(std::string &buffer) {
401  DBG(DBG1, "Schedule receiving datagram from server");
402 
403  // launch async read
404  char b[1 << 16];
405  std::size_t size;
406  set_deadline(opts.timeouts.read);
407  boost::system::error_code ec = asio::error::would_block;
408  socket.async_receive(boost::asio::buffer(b, sizeof(b)),
409  (var(ec) = _1, var(size) = _2));
410  do { ios.run_one();} while (ec == asio::error::would_block);
411 
412  // check whether data was read
413  if (ec) {
414  if (!timeouted) throw io::error_t(err::io_error, ec.message());
415  throw io::error_t(err::timeout, "can't read due to timeout"
416  ": dst=" + addr);
417  }
418 
419  // fill packet
420  packet_t packet(b);
421  DBG(DBG3, "New datagram has been read from server: "
422  "dgram.id=%u, dgram.seq=%d, dgram.count=%d, size=%zd",
423  packet.id, packet.seq, packet.count, size);
424  buffer.append(b + sizeof(packet), size - sizeof(packet));

481 std::string connection_t::read(const std::string &delimiter) {
482  // if no data has been read then read datagram stream from server
483  if (buffer.empty()) fill();
484 
485  // find delimiter
486  std::string::iterator
487  idelim = std::search(buffer.begin(), buffer.end(),
488  delimiter.begin(), delimiter.end());
489  if (idelim == buffer.end())
490  throw io::error_t(err::io_error, "partial input");
491 
492  // prepare result and erase header from buffer
493  std::string result(buffer.begin(), idelim);
494  buffer.erase(buffer.begin(), idelim + delimiter.size());
495  return result;
496 }

498 std::string connection_t::read(std::size_t bytes) {
499  // if no data has been read then read datagram stream from server
500  if (buffer.empty()) fill();
501

504  throw io::error_t(err::io_error, "partial input");
505  std::string result(buffer.data(), bytes);
506  buffer.erase(0, bytes);
507  return result;

510 void connection_t::fill() {
511  // read first packet in datagram stream
512  packet_t packet = socket->receive(buffer);
513  if (packet.seq != 0)
514  throw io::error_t(err::io_error, "first: invalid seq number");
515  if (packet.id != id)
516  throw io::error_t(err::io_error, "first: invalid id");
517 
518  // read all datagrams
519  for (uint16_t i = 1; i < packet.count; ++i) {
520  packet_t next = socket->receive(buffer);
521  if (next.seq != i)
522  throw io::error_t(err::io_error, "next: invalid seq number");
523  if (next.count != packet.count)
524  throw io::error_t(err::io_error, "next: invalid count");
525  if (next.id != id)
526  throw io::error_t(err::io_error, "next: invalid id");
527  }
528 
529  DBG(DBG1, "Whole message: id=%d, size=%zd, buffer=%s",
530  id, buffer.size(), log::escape(buffer).c_str());
531 }
