89  void prepare_serialization() {

91  reserved = htons(reserved);

139 static resp::response_code_t translate_status_to_response(int code) {
140  const std::map<int, resp::response_code_t>::const_iterator
141  it = code_mapping.find(code);
142  if (it == code_mapping.end()) return mc::proto::resp::error;
143  return it->second;
144 }

154 retrieve_command_t::response_t
155 retrieve_command_t::deserialize_header(const std::string &header) const {
156  // reject empty response
157  if (header.empty()) return response_t(resp::empty, "empty response");
158 
159  // parse header && check protocol magic
160  header_t hdr(header);
161  if (hdr.magic != header_t::response_magic)
162  return response_t(resp::unrecognized, "bad magic in response");
163 
164  // check response status
165  if (!hdr.status) {
166  // the response should have uint32_t flags in extras.
167  if (hdr.extras_len != sizeof(uint32_t))
168  return response_t(resp::invalid, "bad extras length");
169  return response_t(0, hdr.body_len, hdr.cas,
170  boost::bind(set_body, _1, _2, _3, hdr.key_len));
171  }
172  return response_t(translate_status_to_response(hdr.status), hdr.body_len);
173 }

213 template <>
214 std::string storage_command_t<true>::serialize(uint8_t code) const {
215  // prepare request
216  aux::check_key(key);
217  header_t hdr(key_len, body_len, extras_len);
218  hdr.opcode = code;
219  if (opts.cas) hdr.cas = opts.cas;
220  hdr.prepare_serialization();
221  std::string result(reinterpret_cast<char *>(&hdr), sizeof(hdr));
222 
223  // append flags
224  uint32_t flags;
225  flags = htonl(opts.flags);
226  result += std::string(reinterpret_cast<char *>(&flags), sizeof(flags));
227 
228  // append expiration; we use flags for expiration too
229  flags = htonl(static_cast<uint32_t>(opts.expiration));
230  result += std::string(reinterpret_cast<char *>(&flags), sizeof(flags));
231 
232  // accomplish request
233  result.append(key);
234  result.append(data);
235  return result;
236 }
