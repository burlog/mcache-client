26 #define rot(x,k) (((x)<<(k)) | ((x)>>(32-(k))))
27 
28 #define mix(a,b,c) \
29 { \
30  a -= c; a ^= rot(c, 4); c += b; \
31  b -= a; b ^= rot(a, 6); a += c; \
32  c -= b; c ^= rot(b, 8); b += a; \
33  a -= c; a ^= rot(c,16); c += b; \
34  b -= a; b ^= rot(a,19); a += c; \
35  c -= b; c ^= rot(b, 4); b += a; \
36 }
37 
38 #define final(a,b,c) \
39 { \
40  c ^= b; c -= rot(b,14); \
41  a ^= c; a -= rot(c,11); \
42  b ^= a; b -= rot(a,25); \
43  c ^= b; c -= rot(b,16); \
44  a ^= c; a -= rot(c,4); \
45  b ^= a; b -= rot(a,14); \
46  c ^= b; c -= rot(b,24); \
47 }

49 uint32_t jenkins ( const void * key, std::size_t length, uint32_t initval )
50 {
51  uint32_t a,b,c; /* internal state */
52 
53  a = b = c = 0xdeadbeef + ((uint32_t)length) + initval;
54 
55  const uint32_t *k = (const uint32_t *)key; /* read 32-bit chunks */
56 
57  /*------ all but last block: aligned reads and affect 32 bits of (a,b,c) */
58  while (length > 12)
59  {
60  a += k[0];
61  b += k[1];
62  c += k[2];
63  mix(a,b,c);
64  length -= 12;
65  k += 3;
66  }
67 
68  switch(length)
69  {
70  case 12: c+=k[2]; b+=k[1]; a+=k[0]; break;
71  case 11: c+=k[2]&0xffffff; b+=k[1]; a+=k[0]; break;
72  case 10: c+=k[2]&0xffff; b+=k[1]; a+=k[0]; break;
73  case 9 : c+=k[2]&0xff; b+=k[1]; a+=k[0]; break;
74  case 8 : b+=k[1]; a+=k[0]; break;
75  case 7 : b+=k[1]&0xffffff; a+=k[0]; break;
76  case 6 : b+=k[1]&0xffff; a+=k[0]; break;
77  case 5 : b+=k[1]&0xff; a+=k[0]; break;
78  case 4 : a+=k[0]; break;
79  case 3 : a+=k[0]&0xffffff; break;
80  case 2 : a+=k[0]&0xffff; break;
81  case 1 : a+=k[0]&0xff; break;
82  case 0 : { return c; } /* zero length strings require no mixing */
83  }
84 
85  final(a,b,c);
86 
87  return c;
88 }
