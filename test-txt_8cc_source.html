44 class validation_connection_t {
45 public:
46  validation_connection_t(const std::string &request,
47  const std::string &header,
48  const std::string &body,
49  int delim = true)
50  : delim(delim), request(request), responses()
51  {
52  responses.push_back(body);
53  responses.push_back(header);
54  }
55 
56  validation_connection_t(const std::string &request,
57  const std::string &header,
58  int delim = true)
59  : delim(delim), request(request), responses()
60  {
61  responses.push_back(header);
62  }
63 
64  void write(const std::string &validate) {
65  if (validate != request)
66  throw validation_connection_error_t("invalid request");
67  }
68 
69  std::string read(const std::string &delimiter) {
70  if (responses.empty())
71  throw validation_connection_error_t("empty response");
72  if (delim && !boost::ends_with(responses.back(), delimiter))
73  throw validation_connection_error_t("invalid delimiter");
74  std::string response = responses.back();
75  responses.pop_back();
76  return response;
77  }
78 
79  std::string read(std::size_t size) {
80  if (responses.empty())
81  throw validation_connection_error_t("empty response");
82  if (size != responses.back().size())
83  throw validation_connection_error_t("invalid size");
84  std::string response = responses.back();
85  responses.pop_back();
86  return response;
87  }
88 
89  bool empty() const { return responses.empty();}
90 
91  bool delim;
92  std::string request;
93  std::vector<std::string> responses;
94 };
