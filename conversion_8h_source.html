159 template <typename protobuf_t>
160 struct cnv<
161  protobuf_t,
162  typename boost::enable_if_c<
163  has_SerializeToString<protobuf_t>::value
164  && has_ParseFromString<protobuf_t>::value
165  >::type
166  > {
167  static protobuf_t as(const std::string &data) {
168  protobuf_t protobuf;
169  if (!protobuf.ParseFromString(data) || !protobuf.IsInitialized())
170  throw error_t(err::bad_argument, "can't deserialize protobuf");
171  return protobuf;
172  }
173  static std::string as(const protobuf_t &protobuf) {
174  std::string data;
175  if (!protobuf.IsInitialized() || !protobuf.SerializeToString(data))
176  throw error_t(err::bad_argument, "can't serialize protobuf");
177  return data;
178  }
